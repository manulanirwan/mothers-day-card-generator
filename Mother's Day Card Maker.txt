<script>
    const canvas = document.getElementById('momCanvas');
    const ctx = canvas.getContext('2d');
    
    let uploadedImage = null;
    let selectedFrameIdx = 0;

    // Define Framework Style Configurations
    const frames = [
        { name: "Elegant Pink", bg: "#FFF5F7", border: "#E91E63", accent: "#F8BBD0" },
        { name: "Golden Blossom", bg: "#FFFDF9", border: "#D4AF37", accent: "#FFF9E6" },
        { name: "Royal Rose", bg: "#FAFAFB", border: "#4A0E17", accent: "#D81B60" },
        { name: "Vintage Minimal", bg: "#Fcfbfa", border: "#7D6B5D", accent: "#EFEBE9" }
    ];

    // Image Upload Handler Engine
    document.getElementById('imageUpload').addEventListener('change', function(e) {
        const reader = new FileReader();
        reader.onload = function(event) {
            const img = new Image();
            img.onload = function() {
                uploadedImage = img;
                drawCard();
            };
            img.src = event.target.result;
        };
        if(e.target.files[0]) {
            reader.readAsDataURL(e.target.files[0]);
        }
    });

    function changeFrame(index) {
        document.querySelector('.mom-item.active').classList.remove('active');
        document.getElementsByClassName('mom-item')[index].classList.add('active');
        selectedFrameIdx = index;
        drawCard();
    }

    // Canvas Render Engine Matrix
    function drawCard() {
        const frame = frames[selectedFrameIdx];
        const greetingText = document.getElementById('userGreeting').value || "";

        // 1. Clear & Render Background Canvas
        ctx.fillStyle = frame.bg;
        ctx.fillRect(0, 0, 600, 600);

        // 2. Compute and Crop User's Custom Image (Center-Cropped Square Shape)
        if (uploadedImage) {
            ctx.save();
            
            // Outer image bounding placeholder properties
            const imgX = 120;
            const imgY = 100;
            const imgSize = 360;

            // Clip into a clean soft-edged rounded rectangle container
            ctx.beginPath();
            ctx.roundRect(imgX, imgY, imgSize, imgSize, 20);
            ctx.clip();

            // Calculate center-cropping ratios aspect variables
            const minSide = Math.min(uploadedImage.width, uploadedImage.height);
            const sx = (uploadedImage.width - minSide) / 2;
            const sy = (uploadedImage.height - minSide) / 2;

            ctx.drawImage(uploadedImage, sx, sy, minSide, minSide, imgX, imgY, imgSize, imgSize);
            ctx.restore();

            // Fine inner gold/pink accent framing around the image container
            ctx.strokeStyle = frame.border;
            ctx.lineWidth = 3;
            ctx.beginPath();
            ctx.roundRect(imgX, imgY, imgSize, imgSize, 20);
            ctx.stroke();
        } else {
            // Informational Box Placeholder when no image is selected
            ctx.fillStyle = frame.accent;
            ctx.beginPath();
            ctx.roundRect(120, 100, 360, 360, 20);
            ctx.fill();

            ctx.fillStyle = frame.border;
            ctx.font = 'italic 16px "Segoe UI", sans-serif';
            ctx.textAlign = 'center';
            ctx.fillText("[ Mom's Photo Will Appear Here ]", 300, 290);
        }

        // 3. Draw Decorative Outer Border Framework Lines
        ctx.strokeStyle = frame.border;
        ctx.lineWidth = 6;
        ctx.strokeRect(25, 25, 550, 550);
        
        ctx.strokeStyle = frame.accent;
        ctx.lineWidth = 2;
        ctx.strokeRect(34, 34, 532, 532);

        // 4. Corner Geometric Graphic Enhancements (Elegant Interlocking Corner Squares)
        ctx.fillStyle = frame.border;
        const offset = [25, 545];
        offset.forEach(x => {
            offset.forEach(y => {
                ctx.fillRect(x == 25 ? x : x-30, y == 25 ? y : y-30, 30, 30);
            });
        });

        // 5. Render Message Text Layout Settings
        ctx.textAlign = 'center';
        ctx.fillStyle = frame.border;
        ctx.font = 'bold 24px "Segoe UI", Arial, sans-serif';

        // Wraps lengthy text sentences cleanly across two lines if required
        const words = greetingText.split(' ');
        let line = '';
        let startY = 510;
        const lineHeight = 32;

        for (let n = 0; n < words.length; n++) {
            let testLine = line + words[n] + ' ';
            let metrics = ctx.measureText(testLine);
            if (metrics.width > 500 && n > 0) {
                ctx.fillText(line, 300, startY);
                line = words[n] + ' ';
                startY += lineHeight;
            } else {
                line = testLine;
            }
        }
        ctx.fillText(line, 300, startY);
    }

    // Monitor Text Inputs dynamically
    document.getElementById('userGreeting').addEventListener('input', drawCard);

    // Secure Local File Download Pipeline
    document.getElementById('downloadMomBtn').addEventListener('click', function() {
        const link = document.createElement('a');
        link.download = `Mothers_Day_Card.png`;
        link.href = canvas.toDataURL('image/png');
        link.click();
    });

    // Default mount initialization
    window.onload = () => { drawCard(); };
    setTimeout(drawCard, 300);
</script>
